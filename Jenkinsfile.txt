pipeline {
    agent any
    stages {
        stage('Clone Repository') {
            steps {
                // Checkout the code from Git
                git url: 'https://github.com/HARINI-V33/java_jenkins.git', branch: 'main'
            }
        }

        stage('Compile Java Code') {
            steps {
                // Compile Java code
                script{
                    bat '''
                        javac -d target Main.java
                }    '''
            }
        }

        stage('Run Java Code') {
            steps {
                // Run the Java application
                 script{
                    bat '''
                        java -cp target Main
                } 
            }
        }
    }

    post {
        always {
            // Archive the output and any other build artifacts if necessary
            archiveArtifacts artifacts: '**/target/*.class', allowEmptyArchive: true
            // Show the build log
            echo 'Build complete.'
        }
    }
}
